# Estates Pay Building Maintenance - On Action Dispatch
@epbm_rebuild_version = 410

on_game_start = {
	on_actions = { epbm_on_initialize }
}

weather_monthly_pulse = {
	on_actions = { epbm_on_global_monthly }
}

monthly_country_pulse = {
	on_actions = { epbm_on_country_monthly }
}

yearly_country_pulse = {
	on_actions = { epbm_on_country_yearly }
}

epbm_on_initialize = {
	effect = {
		epbm_initialize = yes
	}
}

epbm_on_global_monthly = {
	effect = { epbm_global_monthly = yes }
}

epbm_on_country_monthly = {
	trigger = { has_global_variable = epbm_version any_active_estate = { exists = yes } }
	events = { epbm.1 epbm.3 }
}

epbm_on_country_yearly = {
	trigger = {
		has_global_variable = epbm_version
		global_var:epbm_version >= @epbm_rebuild_version
		any_active_estate = { exists = yes }
	}
	events = { epbm.2 }
}
