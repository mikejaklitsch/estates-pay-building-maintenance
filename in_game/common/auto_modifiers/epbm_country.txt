# EPBM country-scoped auto modifiers.

# Raise the building upkeep multiplier from vanilla's 0.2 to 1.0 so the
# goods-cost fraction matches the estate charging calculation.

INJECT:country_base_values = {
	building_upkeep_multiplier = 0.8
}

# Crown-building offset. The estate `power` block reductions above
# reduce the crown's upkeep on *all* buildings, crown buildings
# included, but estates only pay back for non-crown buildings.
# Without this offset, crown buildings end up under-charged.
# Scales by crown_share x (1 - crown_power), which exactly cancels the
# over-discount on the crown's own buildings. The scaler is computed in
# epbm_set_crown_offset_scale (epbm_calculate.txt) once per recalc.

epbm_crown_building_offset = {
	scales_with = epbm_crown_building_offset_scale
	building_upkeep_multiplier = 1.0
}
