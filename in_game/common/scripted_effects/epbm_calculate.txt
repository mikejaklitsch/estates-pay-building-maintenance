# EPBM — calculation engine. Entry point: epbm_calculate_maintenance.
# Estate buildings are charged in full to the owning estate via the
# estate iterator; everything else goes to a shared pool split by
# estate_power, with government buildings routed to crown.
# No pre-registration: PM goods are read directly from the engine.

# ═══════════════════════════════════════════════
# Main calculation entry point. Scope: country.
# Monthly for player, yearly for AI.
# ═══════════════════════════════════════════════

epbm_calculate_maintenance = {
	set_local_variable = { name = epbm_total_cost value = 0 }
	set_local_variable = { name = epbm_crown_total_cost value = 0 }

	epbm_for_each_estate = { effect = epbm_init_estate_var }

	# Estate buildings: iterate per estate, charge directly to each.
	every_estate = {
		limit = { estate_type != estate_type:crown_estate }
		save_temporary_scope_as = epbm_cur_estate

		every_building_owned_by_estate = {
			save_temporary_scope_as = epbm_bldg
			building_type ?= { save_temporary_scope_as = epbm_bt }
			location ?= { market ?= { save_temporary_scope_as = epbm_mkt } }

			epbm_price_building = yes

			if = { limit = { local_var:epbm_bldg_cost > 0 } epbm_route_estate_building_cost = yes }
		}
	}

	# Non-estate buildings: crown and shared pool.
	every_owned_building = {
		limit = {
			is_opened = yes
			NOT = {
				building_type = { is_key_in_global_variable_map = { name = epbm_blacklist target = this } }
			}
			NOT = { building_category = estate_category }
		}

		save_temporary_scope_as = epbm_bldg
		building_type ?= { save_temporary_scope_as = epbm_bt }
		location ?= { market ?= { save_temporary_scope_as = epbm_mkt } }

		epbm_price_building = yes

		if = { limit = { local_var:epbm_bldg_cost > 0 } epbm_route_building_cost = yes }
	}

	epbm_calc_foreign_building_costs = yes

	# S/(S+C): fraction of non-estate pool that is shared (not crown)
	if = {
		limit = { local_var:epbm_total_cost > 0 }
		set_local_variable = {
			name = epbm_shared_fraction
			value = {
				value = local_var:epbm_total_cost
				divide = { value = local_var:epbm_total_cost add = local_var:epbm_crown_total_cost }
			}
		}
	}
	else = { set_local_variable = { name = epbm_shared_fraction value = 0 } }

	epbm_for_each_estate = { effect = epbm_compute_modifier_scale }

	# Country charge factor for scaling raw goods cost
	set_variable = {
		name = epbm_charge_factor
		value = {
			value = epbm_maint_burden
			multiply = epbm_inflation_factor
			divide = epbm_efficiency_divisor
		}
	}

	# Scale shared pool by charge factor
	change_local_variable = { name = epbm_total_cost multiply = var:epbm_charge_factor }

	epbm_split_by_estate_power = yes
	epbm_add_estate_building_costs = yes

	# Crown costs for tooltip: pool share + scaled crown building costs
	set_variable = {
		name = epbm_crown_pool_cost
		value = { value = local_var:epbm_total_cost multiply = "estate_power(estate_type:crown_estate)" }
	}
	set_variable = {
		name = epbm_crown_bldg_cost
		value = { value = local_var:epbm_crown_total_cost multiply = var:epbm_charge_factor }
	}
}

# ═══════════════════════════════════════════════
# Price one building. Scope: building.
# Requires scope:epbm_bldg, scope:epbm_bt, scope:epbm_mkt.
# Finds an input-only PM, gets or computes the per-level base cost,
# then scales by level and building efficiency (employment fraction).
# Sets local_var:epbm_bldg_cost.
# ═══════════════════════════════════════════════

epbm_price_building = {
	set_local_variable = { name = epbm_bldg_cost value = 0 }

	random_production_method_of_building = {
		limit = {
			any_required_goods = { count >= 1 }
			NOT = { exists = produced_goods }
			is_available_for = scope:epbm_bldg.owner
			is_allowed_for = scope:epbm_bldg.owner
			is_visible_for = scope:epbm_bldg.owner
		}

		save_temporary_scope_as = epbm_pm

		prev = {
			set_local_variable = { name = epbm_per_level value = 0 }

			# Check cache on market center
			scope:epbm_mkt = {
				location = {
					if = {
						limit = { has_variable_map = epbm_pm_cache is_key_in_variable_map = { name = epbm_pm_cache target = scope:epbm_pm } }
						set_local_variable = { name = epbm_per_level value = "variable_map(epbm_pm_cache|scope:epbm_pm)" }
					}
				}
			}

			# Cache miss: compute from building_goods_input
			if = {
				limit = { local_var:epbm_per_level <= 0 }

				scope:epbm_pm = {
					every_required_goods = {
						save_temporary_scope_as = epbm_cur_good
						set_local_variable = { name = epbm_price value = "price_in_market(scope:epbm_mkt)" }
						scope:epbm_bldg = { set_local_variable = { name = epbm_qty value = "building_goods_input(scope:epbm_cur_good)" } }
						change_local_variable = { name = epbm_bldg_cost add = { value = local_var:epbm_price multiply = local_var:epbm_qty } }
					}
				}

				# Divide out market access and efficiency to recover base per-level cost
				set_local_variable = { name = epbm_loc_access value = "location.market_access" }
				set_local_variable = { name = epbm_bldg_eff value = "location.building_efficiency(scope:epbm_bt)" }

				if = {
					limit = { local_var:epbm_loc_access > 0 }
					change_local_variable = { name = epbm_bldg_cost divide = local_var:epbm_loc_access }
				}

				if = {
					limit = { local_var:epbm_bldg_eff > 0 }
					change_local_variable = { name = epbm_bldg_cost divide = local_var:epbm_bldg_eff }
				}

				set_local_variable = { name = epbm_per_level value = { value = local_var:epbm_bldg_cost divide = building_level } }

				# Cache the per-level base cost
				scope:epbm_mkt = {
					location = {
						add_to_variable_map = {
							name = epbm_pm_cache
							key = scope:epbm_pm
							value = local_var:epbm_per_level
						}
						add_to_variable_map = {
							name = epbm_bt_costs
							key = scope:epbm_bt
							value = local_var:epbm_per_level
						}

						if = {
							limit = { is_key_in_variable_map = { name = epbm_bt_display target = scope:epbm_bt } }
							remove_from_variable_map = { name = epbm_bt_display key = scope:epbm_bt }
						}

						add_to_variable_map = {
							name = epbm_bt_display
							key = scope:epbm_bt
							value = local_var:epbm_per_level
						}
					}
				}
			}

			# Apply level and efficiency to get actual cost
			set_local_variable = { name = epbm_bldg_eff value = "location.building_efficiency(scope:epbm_bt)" }
			set_local_variable = {
				name = epbm_bldg_cost
				value = {
					value = local_var:epbm_per_level
					multiply = building_level
					multiply = local_var:epbm_bldg_eff
				}
			}
		}
	}
}

# ═══════════════════════════════════════════════
# Route a non-estate building's cost to crown or shared pool.
# ═══════════════════════════════════════════════

epbm_route_building_cost = {
	if = {
		limit = { building_category = government_category }
		change_local_variable = { name = epbm_crown_total_cost add = local_var:epbm_bldg_cost }
	}
	else = { change_local_variable = { name = epbm_total_cost add = local_var:epbm_bldg_cost } }
}

# ═══════════════════════════════════════════════
# Route an estate building's cost to its owning estate.
# Requires scope:epbm_cur_estate from the estate iterator.
# ═══════════════════════════════════════════════

epbm_route_estate_building_cost = {
	epbm_for_each_estate = { effect = epbm_route_to_cur_estate }
}

epbm_route_to_cur_estate = {
	if = {
		limit = { scope:epbm_cur_estate = { estate_type = estate_type:$estate$_estate } }
		change_local_variable = { name = epbm_estate_$estate$ add = local_var:epbm_bldg_cost }
	}
}

# ═══════════════════════════════════════════════
# Foreign buildings: same pricing, crown/shared routing only.
# ═══════════════════════════════════════════════

epbm_calc_foreign_building_costs = {
	every_owned_foreign_building = {
		limit = {
			is_opened = yes
			NOT = {
				building_type = { is_key_in_global_variable_map = { name = epbm_blacklist target = this } }
			}
			NOT = { building_category = estate_category }
		}

		save_temporary_scope_as = epbm_bldg
		building_type ?= { save_temporary_scope_as = epbm_bt }
		location ?= { market ?= { save_temporary_scope_as = epbm_mkt } }

		epbm_price_building = yes

		if = { limit = { local_var:epbm_bldg_cost > 0 } epbm_route_building_cost = yes }
	}
}

# ═══════════════════════════════════════════════
# Country: split the shared pool across estates by estate power
# ═══════════════════════════════════════════════

epbm_split_by_estate_power = {
	if = { limit = { local_var:epbm_total_cost > 0 } epbm_for_each_estate = { effect = epbm_set_pay_from_power } }
	else = { epbm_for_each_estate = { effect = epbm_zero_pay } }
}

# ═══════════════════════════════════════════════
# Country: fold estate-assigned building costs on top of the split
# ═══════════════════════════════════════════════

epbm_add_estate_building_costs = {
	epbm_for_each_estate = { effect = epbm_add_estate_cost }
}
