# Fraction of building maintenance borne by the estate system (0.0-1.0).

epbm_efficiency_divisor = {
	value = 1
	add = "modifier:building_upkeep_efficiency"
	min = 0.5
}

epbm_maint_burden = {
	if = { limit = { has_game_rule = epbm_baseline_50 } value = 0.5 }
	else_if = { limit = { has_game_rule = epbm_baseline_20 } value = 0.2 }
	else = { value = 1.0 }
}

epbm_crown_total_display = {
	value = 0

	if = { limit = { has_variable = epbm_show_crown_pool } add = var:epbm_show_crown_pool }

	if = { limit = { has_variable = epbm_show_crown_bldg } add = var:epbm_show_crown_bldg }
}

# Inflation correction: building_upkeep_multiplier is additive, so
# inflation sits in the same pool as our reductions. Scale our
# contributions by (1 + inflation) so they cover the inflated base.

epbm_inflation_factor = {
	value = 1
	add = inflation

	if = {
		limit = { NOT = { has_game_rule = epbm_yearly_creep_disabled } }
		add = { value = years_since_game_start multiply = 0.001 }
	}
}
