# (1 - crown_power) × inflation. Drives the inflation auto_modifier that
# offsets the non-crown share of vanilla's inflation_impact tax on
# building upkeep. Estates pay their inflated share separately via the
# (1 + inflation) factor in epbm_calculate.txt.

epbm_inflation_times_estate_share = {
	value = 1
	subtract = "estate_power(estate_type:crown_estate)"
	multiply = inflation
}

# Scaler for the crown-building offset auto modifier.
# Source: epbm_set_crown_offset_scale in epbm_calculate.txt, run monthly
# for players and yearly for AI. Value is
#   (crown_upkeep / total_upkeep) × (1 - crown_power) × (1 + inflation)
# which exactly cancels the over-discount that estate `power` blocks and
# the inflation offset apply to the crown's own buildings.

epbm_crown_building_offset_scale = {
	if = { limit = { has_variable = epbm_crown_offset_scale } value = var:epbm_crown_offset_scale }
}
